class Str44 {
                public static void main(String[] args) {
                    System.out.print("Enter the first string: ");
                    String str1 = System.console().readLine();
            
                    System.out.print("Enter the second string: ");
                    String str2 = System.console().readLine();
            
                    boolean contains = str1.contains(str2);
            
                    if (contains) {
                        System.out.println("The first string contains the second string.");
                    } else {
                        System.out.println("The first string does not contain the second string.");
                    }
                }
            }
            
            OUTPUT
            
            
            Enter the first string: mad
            Enter the second string: abc
            The first string does not contain the second string.
