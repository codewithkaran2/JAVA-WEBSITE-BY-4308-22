public class MiddleCharacterFinder {
            public static void main(String[] args) {
                String str1 = "example";
                String str2 = "test";
        
                System.out.println("Middle character(s) of \"" + str1 + "\": " + getMiddleCharacter(str1));
                System.out.println("Middle character(s) of \"" + str2 + "\": " + getMiddleCharacter(str2));
            }
        
            public static String getMiddleCharacter(String str) {
                int length = str.length();
                if (length % 2 == 0) { // Even length
                    return String.valueOf(str.charAt(length / 2 - 1)); // One middle character
                } else { // Odd length
                    return str.charAt(length / 2 - 1) + "" + str.charAt(length / 2); // Two middle characters
                }
            }
        }
