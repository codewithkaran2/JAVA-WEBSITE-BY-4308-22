import javax.swing.*;
        import java.awt.*;
        import java.awt.event.ActionEvent;
        import java.awt.event.ActionListener;
        
        public class ContinuousDots extends JPanel implements ActionListener {
            private int[] xPoints;
            private int[] yPoints;
            private final int DOT_COUNT = 100;
            private final int DOT_SIZE = 5;
        
            public ContinuousDots() {
                xPoints = new int[DOT_COUNT];
                yPoints = new int[DOT_COUNT];
                for (int i = 0; i < DOT_COUNT; i++) {
                    xPoints[i] = (int) (Math.random() * getWidth());
                    yPoints[i] = (int) (Math.random() * getHeight());
                }
        
                Timer timer = new Timer(100, this);
                timer.start();
            }
        
            @Override
            protected void paintComponent(Graphics g) {
                super.paintComponent(g);
                g.setColor(Color.BLUE);
                for (int i = 0; i < DOT_COUNT; i++) {
                    g.fillOval(xPoints[i], yPoints[i], DOT_SIZE, DOT_SIZE);
                }
            }
        
            @Override
            public void actionPerformed(ActionEvent e) {
                for (int i = 0; i < DOT_COUNT; i++) {
                    xPoints[i] = (int) (Math.random() * getWidth());
                    yPoints[i] = (int) (Math.random() * getHeight());
                }
                repaint();
            }
        
            public static void main(String[] args) {
                JFrame frame = new JFrame("Continuous Dots");
                ContinuousDots dotsPanel = new ContinuousDots();
                frame.add(dotsPanel);
                frame.setSize(400, 400);
                frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
                frame.setVisible(true);
            }
        }
