import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class MessageInNewFrame {
    public static void main(String[] args) {
        JFrame mainFrame = new JFrame("Main Frame");
        mainFrame.setSize(300, 200);
        mainFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        mainFrame.setLayout(null);

        JButton showMessageButton = new JButton("Show Message");
        showMessageButton.setBounds(80, 70, 140, 30);
        mainFrame.add(showMessageButton);

        showMessageButton.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                // Create a new frame
                JFrame messageFrame = new JFrame("Message Frame");
                messageFrame.setSize(250, 150);
                messageFrame.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
                messageFrame.setLayout(null);
                
                // Create a label to display the message
                JLabel messageLabel = new JLabel("Hello! This is a message.");
                messageLabel.setBounds(40, 50, 200, 30);
                messageFrame.add(messageLabel);
                
                messageFrame.setVisible(true); // Show the message frame
            }
        });

        mainFrame.setVisible(true); // Show the main frame
    }
}
