import javax.swing.*;
import java.awt.*;
import java.awt.event.KeyAdapter;
import java.awt.event.KeyEvent;

public class EmailTextFieldValidation {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Email ID Validation");
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setSize(300, 150);
        frame.setLayout(new FlowLayout());

        JTextField emailField = new JTextField(20);
        emailField.addKeyListener(new KeyAdapter() {
            public void keyTyped(KeyEvent e) {
                char c = e.getKeyChar();
                String currentText = emailField.getText();
                if (!isValidEmailChar(c, currentText)) {
                    e.consume(); // Ignore the event if the character is not valid for an email
                }
            }
        });

        frame.add(new JLabel("Enter Email ID:"));
        frame.add(emailField);
        frame.setVisible(true);
    }

    private static boolean isValidEmailChar(char c, String currentText) {
        return Character.isLetterOrDigit(c) || c == '@' || c == '.' || c == '_' || c == '-' || c == KeyEvent.VK_BACK_SPACE;
    }
}
