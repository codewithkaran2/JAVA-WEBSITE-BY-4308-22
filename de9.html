import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class FrameSwitchExample {
    public static void main(String[] args) {
        JFrame frame1 = new JFrame("Frame 1");
        JFrame frame2 = new JFrame("Frame 2");

        frame1.setSize(300, 200);
        frame2.setSize(300, 200);
        frame1.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame2.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame1.setLayout(null);
        frame2.setLayout(null);

        JButton switchToFrame2 = new JButton("Go to Frame 2");
        switchToFrame2.setBounds(80, 70, 140, 30);
        frame1.add(switchToFrame2);

        JButton switchToFrame1 = new JButton("Go to Frame 1");
        switchToFrame1.setBounds(80, 70, 140, 30);
        frame2.add(switchToFrame1);

        switchToFrame2.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                frame1.setVisible(false);
                frame2.setVisible(true);
            }
        });

        switchToFrame1.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                frame2.setVisible(false);
                frame1.setVisible(true);
            }
        });

        frame1.setVisible(true);
        frame2.setVisible(false); // Initially hide the second frame
    }
}
