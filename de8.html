import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class SimpleCheckBoxRadioButtonExample {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Simple Checkbox and Radio Button Example");
        frame.setSize(300, 200);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setLayout(new FlowLayout());

        JCheckBox checkBox1 = new JCheckBox("Option A");
        JRadioButton radioButton1 = new JRadioButton("Choice 1");
        JRadioButton radioButton2 = new JRadioButton("Choice 2");
        ButtonGroup radioGroup = new ButtonGroup();
        radioGroup.add(radioButton1);
        radioGroup.add(radioButton2);

        JButton submitButton = new JButton("Submit");
        JLabel label = new JLabel("");

        submitButton.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                StringBuilder selections = new StringBuilder("Selected: ");
                if (checkBox1.isSelected()) selections.append("A ");
                if (radioButton1.isSelected()) selections.append("| Choice 1");
                else if (radioButton2.isSelected()) selections.append("| Choice 2");
                label.setText(selections.toString());
            }
        });

        frame.add(checkBox1);
        frame.add(radioButton1);
        frame.add(radioButton2);
        frame.add(submitButton);
        frame.add(label);
        frame.setVisible(true);
    }
}
