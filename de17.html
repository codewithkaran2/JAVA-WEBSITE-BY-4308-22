import java.applet.Applet;
        import java.awt.*;
        import java.awt.event.ActionEvent;
        import java.awt.event.ActionListener;
        
        public class ChangeAppletBackgroundColor extends Applet {
            Button changeColorButton;
            Color currentColor;
        
            public void init() {
                currentColor = Color.WHITE;
                changeColorButton = new Button("Change Background Color");
                add(changeColorButton);
                changeColorButton.addActionListener(new ActionListener() {
                    public void actionPerformed(ActionEvent e) {
                        currentColor = (currentColor == Color.CYAN) ? Color.WHITE : Color.CYAN;
                        setBackground(currentColor);
                        repaint();
                    }
                });
            }
        
            public void paint(Graphics g) {
                g.drawString("Click the button to change the background color.", 20, 50);
            }
        }
