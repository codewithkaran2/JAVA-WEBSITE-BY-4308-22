class Str32 {
                public static String insertString(String originalString, String stringToInsert, int index) {
                    if (index < 0 || index > originalString.length()) {
                        return originalString;
                    }
                    return originalString.substring(0, index) + stringToInsert + originalString.substring(index);
                }
            
                public static void main(String[] args) {
                    String originalStr = "Hello World!";
                    String strToInsert = " Beautiful";
                    int index = 6;
                    String modifiedStr = insertString(originalStr, strToInsert, index);
            
                    System.out.println("Original String: " + originalStr);
                    System.out.println("Modified String: " + modifiedStr);
                }
            }
            
            OUTPUT
            
            
            Original String: Hello World!
            Modified String: Hello  BeautifulWorld!
