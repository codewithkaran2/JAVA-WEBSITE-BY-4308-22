class Employee {
                private String name;
                private int age;
            
                // Default constructor
                public Employee() {
                    this.name = "XYZ";
                    this.age = 20;
                }
            
            
                public void displayEmployeeInfo() {
                    System.out.println("Name: " + name);
                    System.out.println("Age: " + age);
                }
            }
            class N {
                public static void main(String[] args) {
                    Employee employee = new Employee();
                    employee.displayEmployeeInfo();
                }
            }
            
            OUTPUT
            
            
            Name: XYZ
            Age: 20
