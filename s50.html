class Str50 {
                public static void main(String[] args) {
                    StringBuffer sb = new StringBuffer("Hello");
            
                    System.out.println("Initial string: " + sb.toString());
            
                    // Insert a string at a specific position
                    sb.insert(5, " World");
                    System.out.println("After inserting ' World' at position 5: " + sb.toString());
            
                    // Append a string to the end
                    sb.append("!");
                    System.out.println("After appending '!': " + sb.toString());
            
                    // Delete a substring
                    sb.delete(6, 12);
                    System.out.println("After deleting ' World': " + sb.toString());
            
                    // Replace a substring
                    sb.replace(0, 5, "Hi");
                    System.out.println("After replacing 'Hello' with 'Hi': " + sb.toString());
                }
            }
            
            OUTPUT
            
            
            Initial string: Hello
            After inserting ' World' at position 5: Hello World
            After appending '!': Hello World!
            After deleting ' World': Hello
            After replacing 'Hello' with 'Hi': Hi
