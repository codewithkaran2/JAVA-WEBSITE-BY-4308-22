public class MiddlePointChecker {
            public static void main(String[] args) {
                int num1 = 3;
                int num2 = 5;
                int num3 = 4;
        
                boolean isMiddlePoint = isMiddlePoint(num1, num2, num3);
                System.out.println("Is one number the middle point between the other two? " + isMiddlePoint);
            }
        
            public static boolean isMiddlePoint(int a, int b, int c) {
                return (a + c) / 2.0 == b || (a + b) / 2.0 == c || (b + c) / 2.0 == a;
            }
        }

Output


Is one number the middle point between the other two? true
