public class DigitCounter {
            public static void main(String[] args) {
                int number = 222342;
                int count = countDigitTwo(number);
                System.out.println("The number of digits '2' in " + number + " is: " + count);
            }
        
            public static int countDigitTwo(int num) {
                int count = 0;
                while (num > 0) {
                    if (num % 10 == 2) {
                        count++;
                    }
                    num /= 10;
                }
                return count;
            }
        }

Output


The number of digits '2' in 222342 is: 3
