import java.util.ArrayList;
            import java.util.List;
            
            class MenuItem {
                private String name;
                private double price;
                private double rating;
            
                public MenuItem(String name, double price, double rating) {
                    this.name = name;
                    this.price = price;
                    this.rating = rating;
                }
            
                public String getName() {
                    return name;
                }
            
                public double getPrice() {
                    return price;
                }
            
                public double getRating() {
                    return rating;
                }
            
                @Override
                public String toString() {
                    return "MenuItem{Name='" + name + "', Price=" + price + ", Rating=" + rating + "}";
                }
            }
            
            class Restaurant {
                private List<MenuItem> menu;
            
                public Restaurant() {
                    menu = new ArrayList<>();
                }
            
                // Method to add a menu item
                public void addMenuItem(MenuItem item) {
                    menu.add(item);
                    System.out.println("Added: " + item);
                }
            
                // Method to remove a menu item by name
                public boolean removeMenuItem(String name) {
                    for (MenuItem item : menu) {
                        if (item.getName().equalsIgnoreCase(name)) {
                            menu.remove(item);
                            System.out.println("Removed: " + item);
                            return true;
                        }
                    }
                    System.out.println("Menu item '" + name + "' not found.");
                    return false;
                }
            
                // Method to calculate the average rating of the menu items
                public double calculateAverageRating() {
                    if (menu.isEmpty()) {
                        System.out.println("No menu items available.");
                        return 0.0;
                    }
            
                    double totalRating = 0.0;
                    for (MenuItem item : menu) {
                        totalRating += item.getRating();
                    }
                    double averageRating = totalRating / menu.size();
                    System.out.printf("Average Rating: %.2f%n", averageRating);
                    return averageRating;
                }
            
                // Method to display all menu items
                public void displayMenu() {
                    if (menu.isEmpty()) {
                        System.out.println("No menu items available.");
                    } else {
                        System.out.println("Menu Items:");
                        for (MenuItem item : menu) {
                            System.out.println(item);
                        }
                    }
                }
            }
            
            public class RestaurantSystem {
                public static void main(String[] args) {
                    Restaurant restaurant = new Restaurant();
            
                    // Create some menu items
                    MenuItem item1 = new MenuItem("Pizza", 12.99, 4.5);
                    MenuItem item2 = new MenuItem("Burger", 8.99, 4.7);
                    MenuItem item3 = new MenuItem("Pasta", 10.99, 4.3);
            
                    // Add menu items to the restaurant
                    restaurant.addMenuItem(item1);
                    restaurant.addMenuItem(item2);
                    restaurant.addMenuItem(item3);
            
                    // Display the menu
                    restaurant.displayMenu();
            
                    // Calculate average rating
                    restaurant.calculateAverageRating();
            
                    // Remove a menu item
                    restaurant.removeMenuItem("Burger");
            
                    // Try to remove a menu item that doesn't exist
                    restaurant.removeMenuItem("Sushi");
            
                    // Display the menu after removal
                    restaurant.displayMenu();
            
                    // Recalculate average rating
                    restaurant.calculateAverageRating();
                }
            }
