public class ConsecutiveChecker {
            public static void main(String[] args) {
                int num1 = 5;
                int num2 = 6;
                int num3 = 7;
        
                boolean areConsecutive = areConsecutiveNumbers(num1, num2, num3);
                System.out.println("Are the numbers consecutive? " + areConsecutive);
            }
        
            public static boolean areConsecutiveNumbers(int a, int b, int c) {
                int max = Math.max(a, Math.max(b, c));
                int min = Math.min(a, Math.min(b, c));
                return (max - min == 2) && (a + b + c == 3 * min + 3);
            }
        }

Output


Are the numbers consecutive? true
