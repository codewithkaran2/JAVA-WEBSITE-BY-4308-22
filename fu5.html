public class InvestmentCalculator {
            public static void main(String[] args) {
                double principal = 1000.0; // Initial investment
                double interestRate = 5.0; // Annual interest rate in percent
                int years = 10; // Number of years
        
                double futureValue = calculateFutureValue(principal, interestRate, years);
                System.out.printf("Future investment value: %.2f%n", futureValue);
            }
        
            public static double calculateFutureValue(double principal, double interestRate, int years) {
                return principal * Math.pow(1 + (interestRate / 100), years);
            }
        }

Output


Future investment value: 1628.89
