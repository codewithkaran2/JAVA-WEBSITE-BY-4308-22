class Str38{
                public static void main(String[] args) {
                    String str = "Hello World, this is a test";
                    String swappedCase = "";
            
                    for (int i = 0; i < str.length(); i++) {
                        char c = str.charAt(i);
                        if (Character.isLowerCase(c)) {
                            swappedCase += Character.toUpperCase(c);
                        } else if (Character.isUpperCase(c)) {
                            swappedCase += Character.toLowerCase(c);
                        } else {
                            swappedCase += c;
                        }
                    }
            
                    System.out.println("Original string: " + str);
                    System.out.println("String with swapped case: " + swappedCase);
                }
            }
            
            OUTPUT
            
            
            Original string: Hello World, this is a test
            String with swapped case: hELLO wORLD, THIS IS A TEST
